* Inverter 2 transistors

.include 45nm_bulk.txt
.option TEMP=27C

* Définition des paramètres
.param Vdd = 1.8;
.param Vin = 1.8;
.param L = 4.5e-08;
.param Wn = 3.0571625989695294e-07;
.param Wp = 9.07920147483294e-07;
.param Cload = 1e-12;

* Définition du circuit de l'inverseur
M1 y a vdd vdd pmos L=L W=Wp;            * PMOS
M2 y a 0 0 nmos L=L W=Wn;                * NMOS

* Résistances et tensions
Vdd vdd 0 DC Vdd;                             * Source de tension pour l'alimentation
Vin a 0 PULSE (0 Vdd 0 1p 1p 1u 2u);      * Source de tension d'entrée (peut être ajustée pour simulation)

* Connexion de la sortie
Cl y 0 Cload;                            * Capacité de charge sur la sortie (ajustable selon le besoin)

.tran 1n 2.5u 1p

*mesures
*.meas tran delay TRIG v(a) VAL=0.9999*vdd FALL=1 TARG v(y) VAL=0.5*Vdd RISE=1

*.meas tran i_max MAX I(Vdd)
*.meas tran v_max MAX V(Vdd)
*.meas tran power PARAM=(i_max*v_max)

.control
    run
    gnuplot ../output/results a y i(Vdd)*Vdd
.endc

.end
